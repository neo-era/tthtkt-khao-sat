Cách tạo backend Google Sheets (1 lần)
1. Tạo Google Sheet mới → Tiện ích mở rộng → Apps Script.
2. Dán đoạn mã dưới, deploy dạng Web app (Execute as: Me; Access: Anyone).
3. Copy URL /exec dán vào ô phía trên. 
function doPost(e){
  var sh = SpreadsheetApp.getActiveSpreadsheet().getSheets()[0];
  var rows = JSON.parse(e.postData.contents);
  if(sh.getLastRow()===0){
    sh.appendRow(["STT","Ma","Ten","Duong","SoNha","Phuong","Loai",
      "ViDo","KinhDo","HTChieuSang","HTMangXanh","NguonDien_m",
      "DXChieuSang","DXMangXanh","UuTien","GhiChu","NguoiKS","ThoiGian"]);
  }
  rows.forEach(function(r){ sh.appendRow(r); });
  return ContentService.createTextOutput("OK");
}
